POST /api/auth/register
  Request: { email: string, password: string, displayName: string, token: string }
  Response 201: { token: string, user: { id, email, displayName, username, role } }
  Errors: 400 (missing fields, invalid invite, expired), 409 (email exists)

POST /api/auth/login
  Request: { email: string, password: string }
  Response 200: { token: string, user: { id, email, displayName, username, role } }
  Errors: 401 "Invalid email or password"

POST /api/auth/forgot-password
  Request: { email: string }
  Response 200: { message: "If an account exists, a reset link has been sent" }

POST /api/auth/reset-password
  Request: { token: string, password: string }
  Response 200: { message: "Password reset successful" }
  Errors: 400 (invalid/expired token, short password)

GET /api/health
  Response 200: { status: "ok", timestamp: string }
